int findivisors(int n){
int ideal = 0, sum = 0;
	for (int i=1; i<n; i++){
	if(n%i==0){
	sum = sum + i;
	} else { continue;
	}
	}
	if (sum==n){
	ideal = 1;
}
  return ideal;
}
